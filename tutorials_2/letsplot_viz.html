df2 = df.groupby("Year", as_index=False).agg(avg_hp = ("Horsepower", "mean"), avg_mpg = ("Miles_per_Gallon", "mean")).sort_values("Year")

Code 
(
    ggplot(df2, aes("Year", "avg_hp")) +
    geom_bar(stat = "identity", fill = "darkgreen", alpha = 0.7) +
    geom_line(aes(y = "avg_mpg"), size = 2) +
    geom_point(aes(y = "avg_mpg"), size = 5) +
    scale_x_discrete()
)

Connected Scaterplot

Code 
(
    ggplot(df2, aes("avg_hp", "avg_mpg")) +
    geom_point() +
    geom_segment(
        aes(
        xend = df2["avg_hp"].shift(-1),
        yend = df2['avg_mpg'].shift(-1)
        ),
        arrow = arrow()
    ) +
    geom_text(aes(label = "Year"), nudge_x = 4)
)
